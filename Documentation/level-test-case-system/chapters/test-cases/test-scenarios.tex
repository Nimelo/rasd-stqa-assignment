\chapter{Test scenarios} \label{chp:test-scenarios}
	\begin{testcase}{Passing Configuration Path Test}{Very High}{None}
		{
			\textbf{FN-1.1}
		}
		\term{Objective}{To check if configuration file will load using given path.}
		\term{Inputs}{Configuration file.}
		\term{Steps}
		{
			\begin{enumerate}
				\item Create configuration file
				\item Load configuration file using \emph{-configuration} parameter with the path
				\item Run simulation tool.
			\end{enumerate}
		}
		\term{Output}{Program should notice configuration within given path.}
	\end{testcase}
	
	\begin{testcase}{Passing Report Path Test}{Medium}{None}
		{
			\textbf{FN-1.2}
		}
		\term{Objective}{To check if produced report will be generated in specified location.}
		\term{Inputs}{None}
		\term{Steps}
		{
			\begin{enumerate}
				\item Create configuration file
				\item Load configuration file using \emph{-configuration} parameter.
				\item Load path to report file using \emph{-report} parameter.
				\item Run simulation tool.
			\end{enumerate}
		}
		\term{Output}{Program should produce report within given path.}
	\end{testcase}
